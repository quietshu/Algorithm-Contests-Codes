
program bzoj1559;

var s1,s2,s3,i,j,k,max,p:longint;
    t:array[1..120] of longint;

begin
  read(s1,s2,s3);
  for i:=1 to s1 do
    for j:=1 to s2 do
      for k:=1 to s3 do
        inc(t[i+j+k]);
  max:=0; p:=0;
  for i:=1 to s1+s2+s3 do
    if t[i]>max then
    begin
      max:=t[i];
      p:=i;
    end;
  write(p);
end.
