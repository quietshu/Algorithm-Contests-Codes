#include <iostream>
#include <cstdio>
#define rep(i, n) for(int i = 1; i <= (n); i++)
#define repf(i, l, r) for(int i = (r); i >= (l); i--)
#define MAXM 2000 + 10
#define MAXN 2000 + 10
using namespace std;

int n, m, w[MAXN], f[MAXM], g[MAXM];

int main(){
	scanf("%d%d", &n, &m);
	rep(i, n)
		scanf("%d", &w[i]);
	f[0] = 1;
	rep(i, n)
		repf(j, w[i], m)
			f[j] = (f[j] + f[j - w[i]]) % 10;
	rep(i, n){
		g[0] = 1;
		rep(j, m){
			if(j >= w[i])
				g[j] = f[j] - g[j - w[i]];
			else
				g[j] = f[j];
			if(g[j] < 0)
				g[j] += 10;
			printf("%d", g[j]);
		}
		printf("\n");
	}
	return 0;
}
